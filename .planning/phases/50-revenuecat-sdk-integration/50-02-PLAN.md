1. Create `android/WoofTalk/app/src/main/java/com/wooftalk/EntitlementManager.kt`:
   ```kotlin
   package com.wooftalk

   import com.revenuecat.purchases.CustomerInfo
   import com.revenuecat.purchases.CustomerInfoListener
   import com.revenuecat.purchases.Purchases
   import com.revenuecat.purchases.interfaces.UpdatedCustomerInfoListener
   import kotlinx.coroutines.flow.MutableStateFlow
   import kotlinx.coroutines.flow.StateFlow
   import kotlinx.coroutines.flow.asStateFlow
   import javax.inject.Inject
   import javax.inject.Singleton

   @Singleton
   class EntitlementManager @Inject constructor() : UpdatedCustomerInfoListener {

       private val _isPremium = MutableStateFlow(false)
       val isPremium: StateFlow<Boolean> = _isPremium.asStateFlow()

       private val _isTrialActive = MutableStateFlow(false)
       val isTrialActive: StateFlow<Boolean> = _isTrialActive.asStateFlow()

       private val _dailyTranslationsUsed = MutableStateFlow(0)
       val dailyTranslationsUsed: StateFlow<Int> = _dailyTranslationsUsed.asStateFlow()

       private val _subscriptionTier = MutableStateFlow("free")
       val subscriptionTier: StateFlow<String> = _subscriptionTier.asStateFlow()

       private val _isLoading = MutableStateFlow(false)
       val isLoading: StateFlow<Boolean> = _isLoading.asStateFlow()

       private val _isAuthenticated = MutableStateFlow(false)
       val isAuthenticated: StateFlow<Boolean> = _isAuthenticated.asStateFlow()

       // SDK-06: Unauthenticated users cannot access paywall
       val isReadyToAccessPaywall: StateFlow<Boolean>
           get() = _isAuthenticated.asStateFlow()

       init {
           Purchases.sharedInstance.updatedCustomerInfoListener = this
       }

       // UpdatedCustomerInfoListener — fires on any CustomerInfo change (SDK-04)
       override fun onCustomerInfoUpdated(customerInfo: CustomerInfo) {
           updateFromCustomerInfo(customerInfo)
       }

       private fun updateFromCustomerInfo(customerInfo: CustomerInfo) {
           val proEntitlement = customerInfo.entitlements["pro"]
           val isPremiumActive = proEntitlement?.isActive == true
           _isPremium.value = isPremiumActive

           // Detect trial: active entitlement with willRenew=false or trial period indicators
           val isTrial = isPremiumActive && proEntitlement?.product?.subscriptionPeriod != null
               && customerInfo.activeSubscriptions.any { sub ->
                   // RevenueCat marks trial subscriptions separately
                   customerInfo.allPurchaseDatesByProduct.containsKey(it) &&
                   customerInfo.allExpirationDatesByProduct[it]?.after(java.util.Date()) == true
               }
           _isTrialActive.value = isTrial

           _subscriptionTier.value = when {
               isPremiumActive && !isTrial -> "pro"
               isPremiumActive && isTrial -> "trial"
               else -> "free"
           }
       }

       // D-01/D-02: Call after Supabase auth resolves with auth.uid
       suspend fun logIn(authUid: String) {
           val result = Purchases.sharedInstance.login(authUid)
           updateFromCustomerInfo(result.customerInfo)
           _isAuthenticated.value = true
       }

       // Call on sign-out
       suspend fun logOut() {
           Purchases.sharedInstance.logOut()
           _isAuthenticated.value = false
           _isPremium.value = false
           _isTrialActive.value = false
           _subscriptionTier.value = "free"
       }

       // SDK-05 / Pitfall 2: Force refresh after purchase
       suspend fun refreshEntitlements() {
           _isLoading.value = true
           try {
               val customerInfo = Purchases.sharedInstance.getCustomerInfo()
               updateFromCustomerInfo(customerInfo)
           } catch (e: Exception) {
               // Entitlement check failed — keep current state (D-05: trust cache offline)
           } finally {
               _isLoading.value = false
           }
       }

       // Convenience: check entitlements on foreground / significant events
       suspend fun checkEntitlements() {
           try {
               val customerInfo = Purchases.sharedInstance.getCustomerInfo()
               updateFromCustomerInfo(customerInfo)
           } catch (_: Exception) {
               // D-05: Trust cached CustomerInfo when offline
           }
       }
   }
   ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| App -> RevenueCat API | Client-side SDK communication. API key is public (embedded in APK). |
| RevenueCat -> App (listener) | CustomerInfo delivered via UpdatedCustomerInfoListener. Rooted devices could tamper. |
| App -> Supabase Auth | Auth state determines when logIn is called. Auth token is sensitive. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-50-05 | S (Spoofing) | RevenueCatModule | accept | Client-side API key is public by design. Server-side verification in Phase 51 is the hard gate. |
| T-50-06 | T (Tampering) | EntitlementManager | mitigate | All composables read from single EntitlementManager (D-04). Server-side RLS enforces in Phase 51. |
| T-50-07 | I (Info Disclosure) | EntitlementManager.logIn | accept | auth.uid is a UUID, not PII. |
| T-50-08 | E (Elevation) | EntitlementManager.isPremium | mitigate | Offline cache trusted per D-05. Server-side enforcement in Phase 51 prevents actual bypass. |
</threat_model>
