   ```typescript
   import Purchases from '@revenuecat/purchases-js';
   import type { CustomerInfo } from '@revenuecat/purchases-js';

   let isInitialized = false;

   /**
    * Initialize RevenueCat JS SDK with anonymous/placeholder config.
    * Call logIn(auth.uid) after Supabase auth resolves. (D-01)
    */
   export function initRevenueCat(): void {
     if (isInitialized) return;
     if (typeof window === 'undefined') return; // Skip on server-side

     const apiKey = process.env.NEXT_PUBLIC_REVENUECAT_WEB_API_KEY;
     if (!apiKey) {
       console.warn('NEXT_PUBLIC_REVENUECAT_WEB_API_KEY not set — RevenueCat not initialized');
       return;
     }

     Purchases.configure({ apiKey, appUserID: undefined }); // undefined = anonymous per D-01
     isInitialized = true;
   }

   /**
    * Link RevenueCat identity to Supabase auth.uid.
    * Call after auth resolves and on every launch. (D-02)
    */
   export async function loginRevenueCat(authUid: string): Promise<CustomerInfo | null> {
     if (!isInitialized) {
       initRevenueCat();
     }
     try {
       const { customerInfo } = await Purchases.logIn(authUid);
       return customerInfo;
     } catch (error) {
       console.error('RevenueCat logIn failed:', error);
       return null;
     }
   }

   /**
    * Log out from RevenueCat on Supabase sign-out.
    */
   export async function logoutRevenueCat(): Promise<void> {
     if (!isInitialized) return;
     try {
       await Purchases.logOut();
     } catch (error) {
       console.error('RevenueCat logOut failed:', error);
     }
   }

   /**
    * Force refresh CustomerInfo after purchase. (SDK-05 / Pitfall 2)
    * Must be called immediately after any purchase completes.
    */
   export async function refreshEntitlements(): Promise<CustomerInfo | null> {
     if (!isInitialized) return null;
     try {
       return await Purchases.getCustomerInfo();
     } catch (error) {
       console.error('RevenueCat getCustomerInfo failed:', error);
       return null;
     }
   }

   /**
    * Check if RevenueCat SDK is initialized.
    */
   export function isRevenueCatInitialized(): boolean {
     return isInitialized;
   }
   ```

1. Create `web/src/lib/entitlement-store.ts`:
   ```typescript
   import { create } from 'zustand';
   import type { CustomerInfo } from '@revenuecat/purchases-js';

   interface EntitlementState {
     isPremium: boolean;
     isTrialActive: boolean;
     dailyTranslationsUsed: number;
     subscriptionTier: 'free' | 'trial' | 'pro';
     isLoading: boolean;
     isAuthenticated: boolean;
     error: string | null;

     // SDK-06: Unauthenticated users cannot access paywall
     isReadyToAccessPaywall: boolean;

     // Update state from CustomerInfo
     fromCustomerInfo: (customerInfo: CustomerInfo) => void;
     setLoading: (loading: boolean) => void;
     setAuthenticated: (authenticated: boolean) => void;
     setError: (error: string | null) => void;
     reset: () => void;
   }

   const initialState = {
     isPremium: false,
     isTrialActive: false,
     dailyTranslationsUsed: 0,
     subscriptionTier: 'free' as const,
     isLoading: false,
     isAuthenticated: false,
     error: null,
     isReadyToAccessPaywall: false,
   };

   export const useEntitlementStore = create<EntitlementState>((set, get) => ({
     ...initialState,

     fromCustomerInfo: (customerInfo: CustomerInfo) => {
       const proEntitlement = customerInfo.entitlements.active['pro'];
       const isPremiumActive = proEntitlement !== undefined;

       // Detect trial: active pro entitlement with trial indicators
       // RevenueCat marks trials differently — check if purchase date is within trial window
       const isTrial = isPremiumActive && (() => {
         // A trial is when pro entitlement is active but the user hasn't been charged yet
         // RevenueCat provides this via the active entitlements having a trial period
         const allAccessDates = customerInfo.allPurchaseDatesByProduct;
         const trialProducts = Object.entries(allAccessDates).filter(([productId, date]) => {
           // If product has active entitlement and was purchased recently, could be trial
           return date && customerInfo.entitlements.active['pro'];
         });
         // Simplified: if activeSubscriptions is empty but entitlement is active, likely trial
         return customerInfo.activeSubscriptions.length === 0 && isPremiumActive;
       })();

       set({
         isPremium: isPremiumActive,
         isTrialActive: isTrial,
         subscriptionTier: isPremiumActive ? (isTrial ? 'trial' : 'pro') : 'free',
         isReadyToAccessPaywall: get().isAuthenticated,
       });
     },

     setLoading: (loading: boolean) => set({ isLoading: loading }),

     setAuthenticated: (authenticated: boolean) => set({
       isAuthenticated: authenticated,
       isReadyToAccessPaywall: authenticated,
     }),

     setError: (error: string | null) => set({ error }),

     reset: () => set(initialState),
   }));
   ```

2. Create `web/src/providers/EntitlementProvider.tsx`:
   ```typescript
   'use client';

   import { useEffect } from 'react';
   import { supabase } from '@/lib/supabase';
   import { initRevenueCat, loginRevenueCat, logoutRevenueCat } from '@/lib/revenuecat';
   import { useEntitlementStore } from '@/lib/entitlement-store';
   import Purchases from '@revenuecat/purchases-js';

   export function EntitlementProvider({ children }: { children: React.ReactNode }) {
     const setAuthenticated = useEntitlementStore((s) => s.setAuthenticated);
     const fromCustomerInfo = useEntitlementStore((s) => s.fromCustomerInfo);
     const reset = useEntitlementStore((s) => s.reset);

     useEffect(() => {
       // D-01: Initialize RevenueCat at app mount with anonymous/placeholder
       initRevenueCat();

       // SDK-04: Listen for CustomerInfo updates from RevenueCat
       const unsubscribe = Purchases.addEventListener('customerInfoUpdate', (customerInfo) => {
         fromCustomerInfo(customerInfo);
       });

       // D-02: On auth state change, call logIn(auth.uid) or logOut
       const { data: { subscription } } = supabase.auth.onAuthStateChange(
         async (event, session) => {
           if (event === 'SIGNED_IN' && session?.user?.id) {
             setAuthenticated(true);
             const customerInfo = await loginRevenueCat(session.user.id);
             if (customerInfo) {
               fromCustomerInfo(customerInfo);
             }
           } else if (event === 'SIGNED_OUT') {
             await logoutRevenueCat();
             setAuthenticated(false);
             reset();
           }
         }
       );

       // Check existing session on mount
       supabase.auth.getSession().then(async ({ data: { session } }) => {
         if (session?.user?.id) {
           setAuthenticated(true);
           const customerInfo = await loginRevenueCat(session.user.id);
           if (customerInfo) {
             fromCustomerInfo(customerInfo);
           }
         }
       });

       return () => {
         unsubscribe();
         subscription.unsubscribe();
       };
     }, [setAuthenticated, fromCustomerInfo, reset]);

     return <>{children}</>;
   }
   ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser -> RevenueCat API | Client-side JS SDK over HTTPS. API key is public (NEXT_PUBLIC_ env var). |
| RevenueCat -> App (listener) | CustomerInfo delivered via customerInfoUpdate event. Browser devtools could modify. |
| App -> Supabase Auth | Auth state determines when logIn is called. Session token is sensitive. |

| T-50-12 | E (Elevation) | useEntitlementStore.isPremium | mitigate | Offline cache trusted per D-05. Server-side enforcement in Phase 51 prevents actual bypass. Client is UX layer only. |
