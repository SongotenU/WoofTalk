  <action>
    A. Create 0009_arvr_data_model.sql:
       - ALTER TABLE translation_history ADD COLUMN platform TEXT DEFAULT 'mobile'
         - Allowed values: 'ios', 'android', 'web', 'watch', 'ar_vision', 'vr_quest'
       - ALTER TABLE translation_history ADD COLUMN spatial_position JSONB
         - Format: {"x": float, "y": float, "z": float}
         - Allow NULL for non-AR/VR platforms
       - CREATE TABLE dog_avatars (
           id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
           user_id UUID NOT NULL REFERENCES auth.users(id),
           breed TEXT NOT NULL DEFAULT 'golden',
           color TEXT NOT NULL DEFAULT 'golden',
           accessories JSONB DEFAULT '[]'::jsonb,
           created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
           updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
         )
       - CREATE TABLE user_devices (
           id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
           user_id UUID NOT NULL REFERENCES auth.users(id),
           platform TEXT NOT NULL,
           device_model TEXT,
           os_version TEXT,
           app_version TEXT,
           first_seen TIMESTAMPTZ NOT NULL DEFAULT NOW(),
           last_seen TIMESTAMPTZ NOT NULL DEFAULT NOW(),
           UNIQUE(user_id, platform)
         )
       - UPDATE translation_history SET platform = 'mobile' WHERE platform IS NULL

    B. Create 0010_rls_arvr_tables.sql:
       - Enable RLS on dog_avatars and user_devices
       - CREATE POLICY users_own_dog_avatars ON dog_avatars FOR ALL USING (auth.uid() = user_id)
       - CREATE POLICY users_own_devices ON user_devices FOR ALL USING (auth.uid() = user_id)
       - UPDATE existing RLS policies on translation_history to include platform and spatial_position columns
  </action>
  <acceptance_criteria>
